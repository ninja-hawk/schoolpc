void task_main(intptr_t exinf) {

	while(1){
		pc.baud(9600);
		IrBitField_T irbits;
		zumo.readIr(irbits);
		pc.printf("r:%d, c:%d, l:%d \r\n",
				irbits.right, irbits.center, irbits.left);

		if (irbits.center < 0) {
			if(irbits.left < 0){
				if(irbits.right < 0){
					//8’âŽ~
					zumo.driveTank(0, 0);
					break;
				}
				else{
					//7¶¬

				}
			}
			else{
				if(irbits.right < 0){
					//4‰E¬
					zumo.driveTank(speed, speed*0.4);
				}
				else{
					//3’¼‘å
					zumo.driveTank(speed, speed);
				}

			}
		} else {
			if(irbits.left < 0){
				if(irbits.right < 0){
					//6’¼¬
					zumo.driveTank(speed, speed);
				}
				else{
					//5¶‘å
					zumo.driveTank(speed*0.1, speed);
				}
			}
			else{

					//2‰E‘å
